## SOLUTION:

```python
#!/usr/bin/env python3

def main():
    pi= 3.14
    c= "circumference"
    r= 'radius'

    print("To calculate the " + c + " of a circle, the formula is 2 x " + str(pi) + " x " + r)
    # BONUS
    print(f"To calculate the {c} of a circle, the formula is 2 x {pi} x {r}")

main()
```
